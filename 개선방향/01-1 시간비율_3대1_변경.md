## [3] lib/time-engine.ts — 핵심 계산 + 존 진행 단일 소스

```typescript
export const TIME_RATIO = 3 // 지구 3일 = ToThereOn 1일

export function calculateToThereOnTime(passedDate: string | Date) {
  const passed = new Date(passedDate)
  const now = new Date()
  const earthDaysElapsed = Math.floor(
    (now.getTime() - passed.getTime()) / (1000 * 60 * 60 * 24)
  )
  const currentDay = Math.floor(earthDaysElapsed / TIME_RATIO)

  return {
    currentDay,        // ToThereOn 체류 일수
    earthDaysElapsed,  // 지구 경과 일수
    timeRatio: TIME_RATIO,
  }
}

// 존 진행 — 이 파일이 단일 소스 (zone-manager와 이중화 제거)
export type ZoneId =
  | 'crystal_meadow'
  | 'eternity_forest'
  | 'crystal_lake'
  | 'sunset_hill'

export function getZoneForDay(toThereOnDay: number): ZoneId {
  if (toThereOnDay <= 2)  return 'crystal_meadow'
  if (toThereOnDay <= 9)  return 'eternity_forest'
  if (toThereOnDay <= 29) return 'crystal_lake'
  if (toThereOnDay <= 99) return 'sunset_hill'

  // Day 100+ 순환
  const cycle = (toThereOnDay - 100) % 4
  const zones: ZoneId[] = ['crystal_meadow', 'eternity_forest', 'crystal_lake', 'sunset_hill']
  return zones[cycle]
}

export const ZONE_LOCATIONS: Record<ZoneId, string[]> = {
  crystal_meadow:  ['welcome_plaza', 'butterfly_garden', 'meadow_edge'],
  eternity_forest: ['northern_grove', 'whispering_hollow', 'silver_canopy'],
  crystal_lake:    ['lake_shore', 'quiet_cove', 'deep_waters'],
  sunset_hill:     ['hilltop', 'wind_ridge', 'sunset_terrace'],
}
```

```typescript
import { getPlanConfig } from '@/lib/constants/plans'

async function shouldGenerateFeedToday(pet: Pet, userTier: string): Promise<boolean> {
  const plan = getPlanConfig(userTier)

  // 마지막 피드 생성일 조회
  const { data: lastEvent } = await supabase
    .from('pet_status_events')
    .select('created_at')
    .eq('pet_id', pet.id)
    .order('created_at', { ascending: false })
    .limit(1)
    .single()

  if (!lastEvent) return true // 첫 피드는 항상 생성

  const daysSinceLast = Math.floor(
    (Date.now() - new Date(lastEvent.created_at).getTime()) / (1000 * 60 * 60 * 24)
  )

  return daysSinceLast >= plan.feedIntervalEarthDays
}

// 피드 생성 후: Premium이면 이미지 생성 트리거
async function generateFeedWithImage(pet: Pet, event: StatusEvent, userTier: string) {
  const plan = getPlanConfig(userTier)

  // 텍스트 피드 생성 (기존 로직)
  const feedText = await generateStatusEvent({ pet, event })

  // Premium이면 이미지 추가 생성
  if (plan.imageEnabled) {
    await generatePetImage({
      petId: pet.id,
      context: feedText,         // 피드 내용 기반 장면 생성
      eventId: event.id,
    })
  }

  return feedText
}
```

```typescript
import { calculateToThereOnTime } from '@/lib/time-engine'

async function generateArrivalEvent(pet: Pet) {
  const { currentDay, earthDaysElapsed } = calculateToThereOnTime(pet.passed_date)

  let arrivalType: 'fresh' | 'settling' | 'established'
  let arrivalContext: string

  if (earthDaysElapsed <= 30) {
    arrivalType = 'fresh'
    arrivalContext = 'just arrived, still finding their way'
  } else if (earthDaysElapsed <= 365) {
    arrivalType = 'settling'
    arrivalContext = `has been settling in for ${currentDay} days`
  } else {
    arrivalType = 'established'
    arrivalContext = 'has made this place home long ago'
  }

  await generateStatusEvent({
    pet,
    eventType: 'arrival',
    context: { arrivalType, arrivalContext, toThereOnDay: currentDay }
  })
}

// event-generator.ts의 arrival 프롬프트 분기:
//
// fresh:
//   "The light is on. [Name] has just arrived."
//
// settling:
//   "[Name] has been in ToThereOn World for [N] days.
//    The light was always on. You just found the door."
//
// established:
//   "[Name] has made this place home.
//    All this time, the light was on.
//    You just found the door."
```

## [11] lib/image-generator.ts ⭐ 신규

```typescript
import { getPlanConfig, PREMIUM_IMAGE_RULES } from '@/lib/constants/plans'

interface ImageGenerateParams {
  petId: string
  context: string       // 피드 텍스트 or 편지 텍스트
  eventId?: string      // 피드용
  letterId?: string     // 편지용
}

interface PetVisualProfile {
  species: string    // 'dog' | 'cat' | ...
  breed: string      // 'golden retriever' | ...
  furColor: string   // pets.fur_color 에서 (신규 컬럼)
  size: string       // 'large' | 'medium' | 'small'
  zone: string       // 현재 존
}

export async function generatePetImage(params: ImageGenerateParams) {
  const { petId, context, eventId, letterId } = params

  // 펫 비주얼 정보 조회
  const { data: pet } = await supabase
    .from('pets')
    .select('species, breed, fur_color, size, current_zone')
    .eq('id', petId)
    .single()

  // 이미지 프롬프트 생성
  const prompt = buildImagePrompt({
    species: pet.species,
    breed: pet.breed,
    furColor: pet.fur_color,
    size: pet.size,
    zone: pet.current_zone,
    context,
  })

  // TODO: Kling / NanoBanana Pro API 호출 (스타일 테스트 후 결정)
  // const imageUrl = await klingApi.generate(prompt)
  // const imageUrl = await nanoBananaApi.generate(prompt)

  // 생성된 이미지 저장
  await supabase.from('pet_images').insert({
    pet_id: petId,
    event_id: eventId ?? null,
    letter_id: letterId ?? null,
    prompt,
    image_url: imageUrl,
    created_at: new Date().toISOString(),
  })

  return imageUrl
}

function buildImagePrompt(profile: PetVisualProfile & { context: string }): string {
  const { species, breed, furColor, size, zone, context } = profile

  // 존별 배경 설명
  const zoneBackground: Record<string, string> = {
    crystal_meadow:  'a sunlit meadow with soft crystal-like grass and gentle butterflies',
    eternity_forest: 'an ancient forest with silver-lit canopy and morning mist',
    crystal_lake:    'the edge of a crystal-clear lake in soft golden afternoon light',
    sunset_hill:     'a peaceful hilltop with warm sunset colors and tall grass',
  }

  const background = zoneBackground[zone] ?? 'a peaceful, luminous landscape'

  // 구도: 정면 얼굴 금지, 뒷모습/옆모습/실루엣 중심
  const angles = PREMIUM_IMAGE_RULES.preferredAngles
  const angle = angles[Math.floor(Math.random() * angles.length)]

  return `A ${size} ${breed} with ${furColor} fur, ${angle}, ${background}.
Photorealistic, warm and peaceful atmosphere, soft natural lighting.
No text, no humans, cinematic composition.`

  // 일러스트 스타일로 바꿀 경우:
  // "Studio Ghibli style illustration, soft watercolor, ..."
  // → API 테스트 후 style prefix 교체
}
```
